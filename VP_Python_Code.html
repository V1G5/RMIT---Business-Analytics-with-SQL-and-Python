#Top 5 commodities by household type

top_5 = pd.concat(
    [tmp[tmp.household_type == hh] \
         .sort_values('total_revenue', ascending=False) \
     .head(5) for hh in tmp.household_type.unique()]).reset_index(drop=True)

#Plot of Top 5 commodities by household group
data = []
for d in top_5.commodity.unique():
    tmp1 = top_5[top_5.commodity==d].groupby(['household_type']).agg(revenue=('total_revenue', sum)).reset_index()
    data.append(go.Bar(x=tmp1.household_type, y=tmp1.revenue, name = d))
        
go.Figure(
    data = data,
    layout = go.Layout(
        title ='Top 5 commodities by Household',
        yaxis=dict(
            title='Revenue'
        )
    )
).show(renderer = 'iframe')
